begin

/* This function calculates the gcd of two constants */
/* Prereq: both constants are positive whole numbers. */
num gcd(num x, num y) {
	if (y != 0)
		return gcd(y, x % y);	
	else
		return x;
}

/* Below statements are not in a function so they are executed directly. */

num x, y;
__scan(x);
__scan(y);

num result = gcd(x,y);

__print(result);
__print("\n");

end
