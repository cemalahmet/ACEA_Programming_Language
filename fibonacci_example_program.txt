begin

/* Get how many fibonacci numbers to compute */

num number;
__scan(number);

num last1 = 1, last2 = 0, temp;

/* Compute and print fibonacci numbers */ 
if (number > 0) {
	__print(last2);
	__print("\n");
	number = number - 1;
}

if (number > 0) {
	__print(last1);
	__print("\n");
	number = number - 1;
}

while (number > 0) {
	temp  = last2;
	last2 = last1;
	last1 = temp + last1;
	__print(last1);
	__print("\n");
	number = number - 1;
}

end
